def get_population_number(soup):
      nbr = soup.find(id="field-population")
      if nbr is None:
        return -1
      nbr_pop=nbr.find(class_='subfield-number')
      if nbr_pop is None:
        return -1
      return nbr_pop.text
      
    
    def get_language(soup):
      item = soup.find(id="field-languages")
      if item is None:
        return -1
      lang=item.find(class_="category_data")
      if lang is None:
        return -1
      return lang.text.strip()
